Demo 3 - Override height and label

$('#demo3 .read-more').readmore({
  'clipHeight': function() {
    var maxHeight = 300, clipHeight = 0;

    $(this).children('p').each(function() {
      var height = $(this).outerHeight(true);
      if (clipHeight + height <= maxHeight) {
        clipHeight += height;
      } else {
        return false;
      }
    });

    return (clipHeight > 0 ? clipHeight : maxHeight) + 'px';
  }
});
